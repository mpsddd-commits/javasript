# html에서 그냥 js파일로 사용하면 head에 넣었을때 body 시작 쪽에 끝을 낸다는 onload를 사용해야한다. 왜냐하면 js파일을 보통 스크립트라해서 헤드 안에 넣어놓기때문에 브라우저에 출력이 안된다.
# j쿼리의 경우에는 바디에 온로드같은 형식 없이도 출력이 가능하다.
# ()는 보통 함수형이다 라는 표현
# j쿼리라는 영역이 생기면 브라우져와 서로 다른 영역이 존재한다. 
예) function a(){
  console.log("a");
}//전역함수 -부를 수 있음

$(function(){
  //드디어 jquery영역!!
  console.log("study02.js");
  function b(){
    console.log("b()"); //지역함수 - 브라우저 콘솔로는 부를 수 없음
  }
});
